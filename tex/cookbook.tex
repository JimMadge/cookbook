\begin{recipex}
  \title{Pasta with Feta and Tomatoes}
  \intro{
    This meal is fast and simple, but very tasty and filling.
  }
  \ingredients{
    \component{Pasta}
    \ingredientlist{
      Pasta (Farfalle{,} Conchiglie or Fusilli),
      Cherry Tomatoes,
      Feta,
      Basil,
      Olive Oil
    }
    \component{Teryaki Sauce}
    \ingredientlist{
      Mirin,
      Soy Sauce,
      Sugar
    }
  }
  \method{
    \step Place the pasta in a pan of water with a little salt and bring to
    boil.
    \step While the pasta is cooking prepare the tomatoes and feta. Chop the
    tomatoes into quarters and the feta into cubes of equal size.
    \step When the pasta is al dente, drain the water and return the pasta to
    the pan. Add the olive oil, feta, tomatoes and basil and put over a medium
    heat. Stir and serve immediately when the feta has melted a little and is
    coating the pasta.
  }
\end{recipex}
